60 // Some programs start tons of threads. Let's be stingy with INFO messages.

66 /* We'll satisfy this `promise` inside the thread we are about to spawn -- as soon as it is up and run()ing which
67 * means post()s (etc.) will actually execute from that point on instead of continuing to get queued up.
68 * Then in the current thread we'll briefly await for the `promise` to be satisfied, then return from this ctor;
69 * or if they've passed in their own done_promise (done_promise_else_block not null), then they'll need to
70 * do the satisfaction-awaiting themselves.
71 *
72 * The trial-balloon post()ed task is isn't necessary; post() is formally known to work
73 * across run()ning threads starting and stopping; even with
74 * other tasks pre-queued on the Task_engine (which our doc header contract explicitly says is allowed). It's
75 * a nice trial balloon, and if there's something wrong, it'll block the constructor and make it really obvious;
76 * this has proven to be helpful for debugging/testing in my experience (ygoldfel).
77 *
78 * Update: done_promise has now been extended by a new API arg init_func_or_empty, to indicate in particular
79 * that we've executed init_func_or_empty(). So, while waiting for the subsequent post()ed trial-balloon task
80 * to complete is arguably at best a nicety, nevertheless the fact that can only occur *after* init_func_or_empty()
81 * returns satisfies a *necessity* (to wit: that we don't satisfy done_promise until init_func_or_empty() returns). */
82 unique_ptr<promise<void>> our_done_promise_or_null;
83 promise<void>& done_promise
84 = done_promise_else_block ? *done_promise_else_block
85 : *(our_done_promise_or_null = make_unique<promise<void>>());
86
87 /* As promised in our doc header apply any current verbosity override to the new thread. After all, there's no
88 * opportunity for them to set the override, since there *is* no thread until we start it just now.
89 * For now memorize it; then immediately apply it in the young thread. (If there is no override, that's just
90 * `sev_override == Sev::S_END_SENTINEL`; we need not even track it as a special case.) */
91 const auto sev_override = *(Log_config::this_thread_verbosity_override());

93 m_worker_thread.reset(new Thread([this, // Valid throughout thread { body }.
94 sev_override,
95 nickname = std::move(nickname), // Valid throughout thread { body }.
96 init_func_or_empty = std::move(init_func_or_empty),
97 &done_promise]() // Valid until post({ body }) .set_value()s it.

99 auto const logger_ptr = get_logger();

102 {
103 const auto sev_override_auto = Log_config::this_thread_verbosity_override_auto(sev_override);
104
105 // Now do the pre-loop work.
106
107 Logger::this_thread_set_logged_nickname(nickname, logger_ptr); // This INFO-logs a nice message.

123 post(*m_task_engine, [this,
124 &done_promise]() // done_promise is (by promise/future's nature) alive until we .set_value().

127 {
128 FLOW_LOG_TRACE("First task from Task_qing_thread [" << this << "] executing "
129 "in that thread since not sharing Task_engine object with more threads.");

131 else
132 {
133 FLOW_LOG_TRACE("First task from Task_qing_thread [" << this << "] executing. "
134 "May or may not be in that thread since sharing Task_engine object with more threads.");
135 }
136
137 // Safe to say stuff can be post()ed, possibly to us, as we were post()ed and are executing indeed.
138 done_promise.set_value();
139 FLOW_LOG_TRACE("First task complete.");
140 });

142 * Otherwise it might run from any thread in pool of threads sharing Task_engine; possibly right now concurrently.
143 * Even then, though, subsequent post()s (etc.) might land into this new thread as soon as the following call
144 * starts running in earnest. */
145 } // const auto sev_override_auto = // Restore logging to normal (how it normally is at thread start).
146
147 // @todo boost::asio::io_context::work is deprecated; use its replacement (see io_context boost.asio doc page).
148 Task_engine_work avoid_task_engine_stop(*m_task_engine); // Avoid loop, thread exiting when no pending tasks remain.

150 // Block -- wait for tasks to be posted on this thread's (possibly shared with other threads) Task_engine.
151 m_task_engine->run();
152
153 // Lastly, after everything -- as promised -- re-apply the same Sev override as during startup, for any last msgs.
154 {
155 const auto sev_override_auto = Log_config::this_thread_verbosity_override_auto(sev_override);
156
157 // Now do the post-loop work.
158
159 /* m_task_engine->run() must have exited to have reached here. The only reason this can happen, assuming no
160 * misuse of thread API(s?) by the user, is if stop() (or dtor which runs stop()) ran. Or it threw exception. */
161 FLOW_LOG_INFO("Event loop finished: Task_qing_thread [" << this << "] presumably being stopped! "
162 "Thread exit imminent."); // This is good to keep as INFO.
163 } // const auto sev_override_auto =
164
165 /* Let's discuss handling of various terrible things that can occur inside ->run() which is essentially the thread
166 * body, even though boost.asio is deciding when to run the various pieces of code in that thread body.
167 *
168 * Firstly conditions like SEGV (seg fault -- bad ptr dereference, etc.), FPE (division by zero, etc.),
169 * BUS (like seg fault but with code), etc., may occur and raise a signal via system functionality. That is none
170 * of our concern; to the extent that we could trap such things and do something smart it would apply to threads
171 * not handled by us; so it's not within our purview.
172 *
173 * Secondly an assert() could fail (or, equivalently but much less likely, [std]::abort() could be called directly).
174 * abort() usually raises SIGABRT and, as such, anything we could do about this -- as in the preceding paragraph --
175 * would apply to all threads; so again not in our purview.
176 *
177 * Lastly an exception could be thrown. If caught (within a given task loaded onto *m_task_engine) then that's no
178 * different from handling an error return of a C-style function and wouldn't reach us and is handled within the
179 * task; none of our concern or anyone's concern; it's handled.
180 *
181 * That leaves one last condition: An exception was thrown within a task loaded onto *m_task_engine and *not*
182 * caught within that task. boost.asio shall *not* catch it; hence ->run() above will be emitting an exception.
183 * We can do one of 2 things. We can catch it; or not. If we do catch it there's no way to continue the ->run()
184 * in any fashion that would lead to anything but undefined behavior. We could ->run() again but what happens then
185 * is utterly unpredictable; it's trying to pretend an uncaught exception isn't exceptional. The next least
186 * exceptional behavior for us would be to log something about the exception and then exit the thread the same
187 * was we do above when ->run() returns gracefully (meaning event loop finished gracefully). However, then the
188 * rest of the program would think maybe the thread exited gracefully -- but it didn't -- which again is pretending
189 * an uncaught exception isn't exceptional. (@todo It's conceivable to make this an optional behavior specified
190 * by the *this creator. However even then it would absolutely not be a correct default behavior; so we must
191 * in this discussion decide what *is* correct *default* behavior.)
192 *
193 * We've now whittled down to these possible courses of action in the uncaught-exception-in-a-task scenario.
194 * - Catch it here; then:
195 * -1- Inform the rest of the program in some reasonable way and let them deal with it; e.g., execute some
196 * callback function passed to this ctor, wherein they can do what they want before the thread exits. Or:
197 * -2- Exit the entire program.
198 * -3- Do not catch it here; let it fall through.
199 *
200 * Let's for now not do -1-; not because it's a bad idea (it's a decent idea optionally), but because it doesn't
201 * provide a good default behavior that will do the right thing without the user having to worry about it.
202 *
203 * -2- can be done, basically, either via exit(<non-zero>) or via abort(). exit(<non-zero>) is not good as a
204 * default behavior: Printing the exception's .what() would be decent, but there's no guarantee this would be
205 * flushed to any actual log (e.g., log file) due to the sudden exit() not through main(). In general this wrests
206 * control of the exit() code from the program writer and arbitrarily gives it to Flow. abort(), however, is
207 * fine! After all, an uncaught exception in a thread not managed by us indeed calls abort() through
208 * std::terminate() by default (std::terminate() can also be overridden program-wide). Other than bypassing
209 * std::terminate(), then, it's the same behavior as is generally expected by default. To not bypass
210 * std::terminate() we could instead rethrow the exception.
211 *
212 * -3-, however, is even more direct. If we let it fall through, std::terminate() -- the default behavior -- would
213 * simply be in effect without our inserting ourselves into the situation. If that's all we want (and we've now
214 * made a good case for it) then doing nothing is less entropy then doing something that results in the same thing.
215 *
216 * So -3- is strictly better due to lessened entropy alone -- all else being equal. Is there anything about -2-
217 * that would make "all else" not equal? Yes: we could log .what() first before re-throwing/aborting.
218 * This, however, would come at a major cost: By 1st catching the exception we'd lose the stack trace information:
219 * any generated core would show the stack of the location here in the catch-location, not the original
220 * throw-location. Between printing the exception message and getting the full stack trace in a potential core,
221 * the latter wins in practice by far. At least, by default, we shouldn't be taking that possibility away.
222 *
223 * So that's why we do nothing.
224 *
225 * Last thought: Is there something one can do to *both* get the stack trace in a core file *and* print the
226 * exception .what()? (For that matter, with boost.backtrace and similar, it should be possible to print a stack
227 * trace to the logs as well.) The answer is yes, though it's not on us to do it. One should do such work either
228 * in std::terminate() (by using std::set_terminate()) or, arguably even better, in a global SIGABRT handler.
229 * I am only mentioning it here as opportunistic advice -- again, it's not in our purview, as shown above. */
230 })); // Thread body.
231 // `nickname`, `init_task_or_empty` may now be hosed.
232
233 if (done_promise_else_block)
234 {
235 FLOW_LOG_TRACE("Thread started -- confirmation that it is up shall be signalled to caller through a `promise`.");
236 }
237 else
238 {
239 FLOW_LOG_TRACE("Thread started -- awaiting confirmation it is up.");
240 done_promise.get_future().wait();
241 FLOW_LOG_TRACE("Confirmed thread started -- Task_qing_thread [" << this << "] ready.");
242 }

256void Task_qing_thread::stop()
257{
258 /* Subtlety: stop() is to be a harmless no-op if called twice. We therefore perform an explicit check for it
259 * and return if detected. This isn't strictly necessary; Task_engine::stop() and Thread::join() do work if called
260 * again after calling them once, and they just no-op, as desired. We do it because:
261 * - It's cleaner: one less code path to have to reason about in terms of safety.
262 * - Less spurious/verbose logging.
263 * - It's a tiny bit faster.
264 *
265 * This isn't a mere corner case; the destructor calls us even if stop() was called explicitly earlier. */
266 const bool redundant = !m_worker_thread->joinable(); // A/k/a "not-a-thread" <=> "not joinable."
267
268 if (redundant)
269 {
270 FLOW_LOG_TRACE("Task_qing_thread [" << this << "]: Not waiting for worker thread "
271 "[T" << m_worker_thread->get_id() << "] to finish, as it was already stop()ped earlier.");
272 }
273 else
274 {
275 FLOW_LOG_INFO("Task_qing_thread [" << this << "]: Waiting for worker thread "
276 "[T" << m_worker_thread->get_id() << "] to finish, as it was not stop()ped earlier.");
277 }
278 // Ensure we understand properly the relationship between Task_engine::stopped() and this thread's existence.
279 assert((!m_own_task_engine) // In this case, as promised, it's up to them to stop m_task_engine at some point.
280 || (m_task_engine->stopped() == redundant)); // && m_own_task_engine)
281 if (redundant)
282 {
283 return;
284 }
285 // else OK; this is the first stop() call. Do as we promised.
286
287 if (m_own_task_engine)
288 {
289 m_task_engine->stop(); // Trigger our owned Task_engine to execute no futher tasks and hence exit run().
290 }
291 // else { Caller will have to do something similar before the following statement is able to return. }
292
293 m_worker_thread->join();
294 // Mirror the top of this method.
295 assert(!m_worker_thread->joinable());
296 assert((!m_own_task_engine) || m_task_engine->stopped());
297
298 FLOW_LOG_TRACE("Task_qing_thread [" << this << "]: Thread stopped/joined; stop() return imminent.");
299} // Task_qing_thread::stop()
