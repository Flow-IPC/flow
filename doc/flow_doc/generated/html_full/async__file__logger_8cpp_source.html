29Async_file_logger::Async_file_logger(Logger* backup_logger_ptr,
30 Config* config, const fs::path& log_path,
31 bool capture_rotate_signals_internally) :
32 /* Set up the logging *about our real attempts at file logging*, probably to a Simple_ostream_logger/cout;
33 * or to null (meaning no such logging). Either way we will still attempt to actually log user-requested messages
34 * to log_path! */
35 Log_context(backup_logger_ptr, Flow_log_component::S_LOG),
36 m_config(config), // Save pointer, not copy, of the config given to us. Hence thread safety is a thing.
37 // Any I/O operations done here are the only ones not done from m_async_worker thread (until maybe dtor).
38 m_serial_logger(boost::movelib::make_unique<Serial_file_logger>(get_logger(), m_config, log_path)),
39 /* ...Speaking of which: start the worker thread right now; synchronously; return from this ctor once it has
40 * certified it is up. It will log (to *backup_logger_ptr if not null) about starting up right here, too, which
41 * in practice would be the bulk of INFO-or-less-verbose log lines from us in my experience. */
42 m_async_worker(backup_logger_ptr, util::ostream_op_string("Async_file_logger[", this, ']')),
43 /* Set up a signal set object; this is a no-op until we .add() signals to it (which we may or may not do).
44 * Whether we do or not is more significant than merely whether whatever handler we'd later register
45 * via m_signal_set.async_wait() will be called; if we .add() zero signals, then IF some non-boost.asio
46 * signal handler is currently registered (such as some default OS handler; or the user's non-boost.asio handler)
47 * (or will be registered in the future) will continue to work undisturbed. If, however, we .add() one or more
48 * signals (or, equivalently, list 1 or more signal numbers right here in the constructor call),
49 * then we will REPLACE such non-boost.asio handlers with boost.asio's mechanism. (This behavior is
50 * explicitly documented in boost.asio docs.) Therefore we must be careful not to mindlessly .add() handler(s),
51 * and/or (equivalently) list 1 or more signal numbers in this constructor call here. The choice will be left to
52 * later code which will .add() or not .add() deliberately. (Copy/paste warning: This cmnt was copy-pasted from some
53 * vaguely similar code in flow::net_flow; but the 2 modules are barely related if at all, so....) */
54 m_signal_set(*(m_async_worker.task_engine()))
55{
56 m_async_worker.start();
57
58 FLOW_LOG_INFO("Async_file_logger [" << this << "]: "
59 "Log-writing worker thread(s) have started around now; ready to work.");
60
61 // Immediately open file in worker thread. Could be lazy about it but might as well log about any issue ASAP.
62 log_flush_and_reopen(false); // Do it synchronously for good measure; doesn't matter but the reduced entropy is nice.
63
64 if (capture_rotate_signals_internally)
65 {
66 FLOW_LOG_INFO("Setting up internal log-rotate-reopening signal handler. "
67 "CAUTION! User program MUST avoid using non-boost::asio::signal_set signal handling! "
68 "If it does use non-boost.asio, behavior is undefined.");
69
70 // Add the classic log-rotation-inducing signal numbers.
71 m_signal_set.add(SIGHUP);
72
73 /* At this point, receiving those signals will NOT do whatever default behavior would happen -- but I believe
74 * that unlike for SIGTERM/etc. the default behavior is no-op. Before actually reporting successful initialization
75 * by exiting constructor, set up the handler that'll be called upon receiving the signals. */
76
77 /* this->impl_log_flush_and_reopen() will be called on signal (or error).
78 * Note that that function's contract (from its doc comment) is it must execute in m_async_worker (see
79 * m_signal_set() init above in the init section of ctor).
80 * Indeed boost::asio::io_service semantics guarantee it'll run in m_async_worker (not some
81 * no-man's-land signal handler thread of execution, as one might fear could be the case) for the same reason
82 * the various post()ed tasks will run in m_async_worker. */
83 m_signal_set.async_wait([this](const Error_code& sys_err_code, int sig_number)
84 {
85 on_rotate_signal(sys_err_code, sig_number);
86 });
87 } // if (capture_rotate_signals_internally)
88 /* else if (!capture_rotate_signals_internally)
89 * {
90 * Do NOT .add() anything; and don't async_wait() anything. As noted in comment at m_signal_set construction time,
91 * .add() does more than make it possible to .async_wait(). It also replaces any default OS or user's own
92 * non-boost.asio signal handling machinery with boost.asio's signal_set machinery. That can be quite draconian,
93 * so user must specifically set that option to true. If it's false (in all siblings of *this in entire app),
94 * then whatever signal handling machinery the user wants to set up for themselves (or leave at OS's
95 * discretion) will remain undisturbed. By the way, of course, they can use boost.asio machinery themselves too;
96 * it's just that doing so would still work even if capture_rotate_signals_internally were true, so that's
97 * not the dangerous scenario. (Copy/paste warning: This cmnt was copy-pasted from some
98 * vaguely similar code in flow::net_flow; but the 2 modules are barely related if at all, so....)
99 * } */
100} // Async_file_logger::Async_file_logger()

102Async_file_logger::~Async_file_logger() // Virtual.
103{
104 using async::Synchronicity;
105
106 FLOW_LOG_INFO("Async_file_logger [" << this << "]: Deleting. Worker thread will flush "
107 "output if possible; then we will proceed to shut down.");
108
109 /* Could do this from current thread (in fact if we simply deleted the following statement, that is what would
110 * happen); that's still non-concurrent with other calls, as Serial_file_logger requires.
111 * I (ygoldfel) have a soft spot for being able to say only 1 thread touches the file after ctor; maybe that's
112 * silly. Not a to-do to change it IMO. */
113 m_async_worker.post([this]() { m_serial_logger.reset(); },
114 Synchronicity::S_ASYNC_AND_AWAIT_CONCURRENT_COMPLETION);
115
116 // Now thread will exit and be joined, etc.
117}

119void Async_file_logger::on_rotate_signal(const Error_code& sys_err_code, int sig_number)
120{
121 // We are in m_async_worker thread.
122
123 if (sys_err_code == boost::asio::error::operation_aborted)
124 {
125 return; // Stuff is shutting down; just get out.
126 }
127 // else
128
129 FLOW_LOG_INFO("Async_file_logger [" << this << "]: "
130 "Internal log-rotate handler executed with signal number [" << sig_number << "].");
131
132 if (sys_err_code)
133 {
134 // This is odd, but there's no need to freak out about anything else. Just log and get out.
135 FLOW_ERROR_SYS_ERROR_LOG_WARNING();
136 FLOW_LOG_WARNING("Async_file_logger [" << this << "]: "
137 "Internal signal handler executed with an error indicator. Strange! "
138 "Ignoring and continuing other operation.");
139 }
140 else
141 {
142 m_serial_logger->log_flush_and_reopen();
143 }
144
145 // Wait for it again, or else a rotation would only occur on the first signal but not thereafter.
146 m_signal_set.async_wait([this](const Error_code& sys_err_code, int sig_number)
147 {
148 on_rotate_signal(sys_err_code, sig_number);
149 });
150} // Async_file_logger::on_rotate_signal()

152void Async_file_logger::do_log(Msg_metadata* metadata, util::String_view msg) // Virtual.
153{
154 using util::Blob;
155 using util::String_view;
156 using boost::asio::const_buffer;
157
158 assert(metadata);
159
160 /* Key points about how the asynchronicity/logging/queueing works: We aim to return synchronously ASAP and leave the
161 * potentially blocking (like if hard drive has to turn on from sleep) I/O ops to the worker thread asynchronously.
162 * There are 2 pieces of data, *metadata and `msg`, to deal with, both of which essentially must be available when
163 * we in fact write to file via m_serial_logger. By contract of do_log() and logs_asynchronously()==true:
164 * - We are to NOT copy *metadata; and therefore we are to `delete metadata` when done with it; i.e., after
165 * m_serial_logger->do_log() returns having written to file-system.
166 * - We MUST copy `msg` so we can use it asynchronously; we are to therefore free the copy ourselves once
167 * m_serial_logger->do_log() returns. (In reality this is all because `msg` is a shared thread-lcl thing that is
168 * reused in every subsequent and preceding FLOW_LOG_WARNING/etc. invocation in this thread. We technically don't
169 * know or care about that reality in here; only the *result* wherein we cannot use `msg` after do_log()
170 * synchronously returns.) */
171
172 /* Use our binary Blob utility container to have tight control over copying/allocation. Namely:
173 * Allocate (and do not even zero-initialize!) the exact # of bytes needed in the new Blob.
174 * Copy (memcpy() likely) those exact bytes including the NUL into that area.
175 * Copy-less-ly (via std::move()) transfer this Blob into the lambda executed in the worker thread. */
176 Blob msg_copy_blob_to_move(0);
177 msg_copy_blob_to_move.assign_copy(const_buffer(msg.data(), msg.size()));
178
179 m_async_worker.post([this, metadata,
180 msg_copy_blob = std::move(msg_copy_blob_to_move)]()
181 {
182 /* We are in m_async_worker thread, as m_serial_logger requires.
183 * *metadata and msg_copy_blob are to be freed when done. */
184 m_serial_logger->do_log(metadata, String_view(reinterpret_cast<const char*>(msg_copy_blob.const_data()),
185 msg_copy_blob.size()));
186
187 // As promised, delete this (which was never copied at all); and msg_copy_blob is freed once this {} exits soon.
188 delete metadata;
189 }); // m_async_worker.post()
190} // Async_file_logger::do_log()

192void Async_file_logger::log_flush_and_reopen(bool async)
193{
194 using async::Synchronicity;
195
196 // We are in some unspecified user thread that isn't m_async_worker.
197
198 FLOW_LOG_INFO("Async_file_logger [" << this << "]: Non-worker (user) thread "
199 "requested [" << (async ? "asynchronous" : "synchronous") << "] file flush/close (if needed) and "
200 "re-open, such as for rotation or at initialization.");
201
202 m_async_worker.post([this]() { m_serial_logger->log_flush_and_reopen(); },
203 async ? Synchronicity::S_ASYNC : Synchronicity::S_ASYNC_AND_AWAIT_CONCURRENT_COMPLETION);
204}
