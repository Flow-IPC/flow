28/**
29 * The boost.system category for errors returned by the `net_flow` Flow module. Think of it as the polymorphic
30 * counterpart of error::Code, and it kicks in when, for `flow::Error_code ec`, something
31 * like `ec.message()` is invoked.
32 *
33 * Note that this class's declaration is not available outside this translation unit (.cpp
34 * file), and its logic is accessed indirectly through standard boost.system machinery
35 * (`Error_code::name()` and `Error_code::message()`). This class enables those basic
36 * features (e.g., `err_code.message()` returns the string description of the error `err_code`) to
37 * work.
38 */
39class Category :
40 public boost::system::error_category
41{
42public:
43 // Constants.
44
45 /// The one Category.
46 static const Category S_CATEGORY;
47
48 // Methods.
49
50 /**
51 * Implements superclass API: returns a `static` string representing this `error_category` (which,
52 * for example, shows up in the `ostream` representation of any Category-belonging
53 * flow::Error_code).
54 *
55 * @return A `static` string that's a brief description of this error category.
56 */
57 const char* name() const noexcept override;
58
59 /**
60 * Implements superclass API: given the integer error code of an error in this category, returns a description of that
61 * error (similarly in spirit to `std::strerror()`). This is the "engine" that allows `ec.message()`
62 * to work, where `ec` is a flow::Error_code with an error::Code value.
63 *
64 * @param val
65 * Error code of a Category error (realistically, an error::Code `enum` value cast to
66 * `int`).
67 * @return String describing the error.
68 */
69 std::string message(int val) const override;
70
71private:
72 // Constructors.
73
74 /// Boring constructor.
75 explicit Category();
76}; // class Category
77
78// Static initializations.
79
80const Category Category::S_CATEGORY;
81/* ^-- @todo Note that things will go bad, if something triggers an error before this is actually instantiated, AND
82 * the error is accessed in such a way as to need the error_category -- e.g., accessing its error message for
83 * logging. This isn't very likely, but if some other static initialization code triggers an error, then there
84 * is a random chance it would be before this static initialization, hence the likely problem of the seg-fault
85 * type. We can fight this by using a local static variable trick, but then that isn't thread-safe by default.
86 * This is a well known C++ dilemma without a perfect solution. In this case we think it's reasonable to
87 * expect Flow to not be used until main() runs, so we'll live with the risk of this. Revisit/reevaluate,
88 * especially if we upgrade to C++0x. Update: We are now compiling with C++11. http://cppreference.com does
89 * not mention any relevant improvements in 11 or 14 or 17. So we are stuck for now. */

93Error_code make_error_code(Code err_code)
94{
95 /* Assign Category as the category for net_flow::error::Code-cast error_codes;
96 * this basically glues together Category::name()/message() with the Code enum. */
97 return Error_code(static_cast<int>(err_code), Category::S_CATEGORY);
98}

107std::string Category::message(int val) const // Virtual.
108{
109 using std::string;
110
111 // KEEP THESE STRINGS IN SYNC WITH COMMENT IN net_flow_error.hpp ON THE INDIVIDUAL ENUM MEMBERS!
112
113 /* Just create a string (unless compiler is smart enough to do this only once and reuse later)
114 * based on a static char* which is rapidly indexed from val by the switch() statement.
115 * Since the error_category interface requires that message() return a string by value, this
116 * is the best we can do speed-wide... but it should be fine.
117 *
118 * Some error messages can be fancier by specifying outside values (e.g., see
119 * S_INVALID_SERVICE_PORT_NUMBER). */
120 switch (static_cast<Code>(val))
121 {
122 case Code::S_NODE_NOT_RUNNING:
123 return "Node not running.";
124 case Code::S_CANNOT_CONNECT_TO_IP_ANY:
125 return "Cannot ask to connect to \"any\" IP address. Use specific IP address.";
126 case Code::S_INVALID_SERVICE_PORT_NUMBER:
127 {
128 string str;
129 util::ostream_op_to_string(&str,
130 "Flow port number is not in the valid service port number range [",
131 Port_space::S_FIRST_SERVICE_PORT, ", ",
132 Port_space::S_FIRST_SERVICE_PORT + Port_space::S_NUM_SERVICE_PORTS - 1, "].");
133 return str;
134 }
135 case Code::S_PORT_TAKEN:
136 return "Flow port already reserved.";
137 case Code::S_INTERNAL_ERROR_PORT_NOT_TAKEN:
138 return "Internal error: Tried to return Flow port which had not been reserved.";
139 case Code::S_OUT_OF_PORTS:
140 return "No more ephemeral Flow ports available.";
141 case Code::S_INTERNAL_ERROR_PORT_COLLISION:
142 return "Internal error: Ephemeral port double reservation allowed.";
143 case Code::S_CONN_RESET_BAD_PEER_BEHAVIOR:
144 return "Connection reset because of unexpected/illegal behavior by the other side.";
145 case Code::S_CONN_REFUSED:
146 return "Other side refused connection.";
147 case Code::S_INTERNAL_ERROR_SYSTEM_ERROR_ASIO_TIMER:
148 return "Internal error: System error: Something went wrong with boost.asio timer subsystem.";
149 case Code::S_CONN_TIMEOUT:
150 return "Other side did not complete connection handshake within the allowed time; perhaps no one is listening.";
151 case Code::S_CONN_RESET_BY_OTHER_SIDE:
152 return "Other side reset an established connection.";
153 case Code::S_CONN_RESET_TOO_MANY_REXMITS:
154 return "Connection reset because a packet has been retransmitted too many times.";
155 case Code::S_SEQ_NUM_ARITHMETIC_FAILURE:
156 return "Other side has sent packets with inconsistent sequence numbers.";
157 case Code::S_SEQ_NUM_IMPLIES_CONNECTION_COLLISION:
158 return "Other side has sent packet with sequence number that implies a port collision between "
159 "two connections over time.";
160 case Code::S_NODE_SHUTTING_DOWN:
161 return "Node shutting down.";
162 case Code::S_USER_CLOSED_ABRUPTLY:
163 return "User code on this side abruptly closed connection; other side may be informed of this.";
164 case Code::S_EVENT_SET_CLOSED:
165 return "Attempted operation on an event set, when that event set was closed.";
166 case Code::S_EVENT_SET_IMMUTABLE_WHEN_WAITING:
167 return "Attempted to write to an event set, while a wait operation was pending on that event set.";
168 case Code::S_EVENT_SET_EVENT_ALREADY_EXISTS:
169 return "Attempted to add an event into an event set, but that event already exists.";
170 case Code::S_EVENT_SET_EVENT_DOES_NOT_EXIST:
171 return "Attempted to work with an event that does not exist in the event set.";
172 case Code::S_EVENT_SET_DOUBLE_WAIT_OR_POLL:
173 return "Attempted to wait on or poll an event set while already waiting on that event set.";
174 case Code::S_EVENT_SET_NO_EVENTS:
175 return "Attempted to wait on an event set without specifying event on which to wait.";
176 case Code::S_EVENT_SET_RESULT_CHECK_WHEN_WAITING:
177 return "Attempted to check wait results while still waiting.";
178 case Code::S_CONN_METADATA_TOO_LARGE:
179 return "During connection user supplied metadata that is too large.";
180 case Code::S_STATIC_OPTION_CHANGED:
181 return "When setting options, tried to set an unchangeable (static) option.";
182 case Code::S_OPTION_CHECK_FAILED:
183 return "When setting options, at least one option's value violates a required condition on that option.";
184 case Code::S_WAIT_INTERRUPTED:
185 return "A blocking synchronous or asynchronous operation was interrupted, such as by a signal.";
186 case Code::S_WAIT_USER_TIMEOUT:
187 return "A blocking synchronous or asynchronous operation reached user-specified timeout without succeeding.";
188 }
189 assert(false);
190 return "";
191} // Category::message()
