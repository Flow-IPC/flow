26Duration_set operator-(const Time_pt_set& to, const Time_pt_set& from)
27{
28 Duration_set result;
29 // Could use some algorithm probably but speed is important, and I just don't care to risk it here for some syn-sugar.
30 for (size_t clock_type_idx = 0; clock_type_idx != result.m_values.size(); ++clock_type_idx)
31 {
32 result.m_values[clock_type_idx]
33 = to.m_values[clock_type_idx] - from.m_values[clock_type_idx]; // Can be negative; allowed.
34 }
35 return result;
36}

78std::ostream& operator<<(std::ostream& os, Clock_type clock_type)
79{
80 switch (clock_type)
81 {
82 case Clock_type::S_REAL_HI_RES: return os << "real_hi";
83 case Clock_type::S_CPU_USER_LO_RES: return os << "cpu_usr_lo";
84 case Clock_type::S_CPU_SYS_LO_RES: return os << "cpu_sys_lo";
85 case Clock_type::S_CPU_TOTAL_HI_RES: return os << "cpu_total_hi";
86 case Clock_type::S_CPU_THREAD_TOTAL_HI_RES: return os << "cpu_thread_total_hi";
87 case Clock_type::S_END_SENTINEL: assert(false && "Should never print sentinel.");
88 }
89 assert(false && "Should be unreachable; compiler should warn if incomplete switch.");
90 return os;
91}

93std::ostream& operator<<(std::ostream& os, const Duration_set& duration_set)
94{
95 using std::flush;
96 using flow::util::ostream_op_to_string;
97 using flow::util::String_ostream;
98 using boost::chrono::microseconds;
99 using boost::chrono::nanoseconds;
100 using boost::chrono::round;
101
102 // @todo Skip things not in m_which_clocks.
103
104 for (size_t clock_type_idx = 0; clock_type_idx != duration_set.m_values.size(); ++clock_type_idx)
105 {
106 const auto& dur = duration_set.m_values[clock_type_idx];
107 os << Clock_type(clock_type_idx) << ": ";
108 if (dur.count() == 0)
109 {
110 os << '0'; // No units needed.
111 }
112 else
113 {
114 os << round<nanoseconds>(dur) << "=~" << round<microseconds>(dur); // Prints units too.
115 }
116 if ((clock_type_idx + 1) != duration_set.m_values.size())
117 {
118 os << " | ";
119 }
120 }
121 return os << flush;
122}
