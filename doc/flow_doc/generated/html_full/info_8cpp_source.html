67void Peer_socket_receive_stats::output(std::ostream* os) const
68{
69 using boost::chrono::duration;
70 using boost::chrono::duration_cast;
71 using boost::chrono::seconds;
72 using Float_seconds = duration<double, seconds::period>;
73
74 // Use a floating point representation for socket lifetime so that it can be printed in floating-point seconds.
75 const Float_seconds lifetime = duration_cast<Float_seconds>(Fine_clock::now() - m_init_time);
76
77 /* Compute the drop rate based on the # of bytes in unreceived packets that had been unreceived
78 * for so long we consider them to be forever gone; this is divided by that plus the # of data
79 * bytes that were received normally. The latter does include any data we dropped due to
80 * insufficient buffer space, as we're trying to measure the network drop rate which is, loosely,
81 * a function of the network and the congestion control algorithm's behavior
82 * on the sender side. */
83 const double den = double(m_presumed_dropped_data_size + m_good_data_size);
84 const double drop_pct
85 = (den == double(0))
86 ? 0
87 : (double(m_presumed_dropped_data_size) * double(100) / den);
88
89 *os
90 << "[rcv] format of sizes: [bytes|packets]\n"
91 "[rcv] socket_lifetime [" << lifetime << "]\n"
92 "[rcv] lifetime_goodput_delivered ["
93 << util::to_mbit_per_sec<Float_seconds>(double(m_good_data_delivered_size) / lifetime.count()) << " Mbit/s]\n"
94 "[rcv] total_data [" << m_total_data_size << '|' << m_total_data_count << "]\n"
95

97

104 "first_queued [" << m_good_data_first_qd_size << '|' << m_good_data_first_qd_count << "]\n"
105
106 "[rcv] queued [" << (m_good_data_accepted_size - m_good_data_delivered_size)
107 << '|' << (m_good_data_accepted_count - m_good_data_delivered_count) << "]\n"

111 << '|' << m_good_data_dropped_buf_overflow_count << "]\n"

124 "[rcv] total_to_send_acks_data [" << m_total_to_send_acks_data_size

190void Peer_socket_send_stats::output(std::ostream* os) const
191{
192 using boost::chrono::duration;
193 using boost::chrono::duration_cast;
194 using boost::chrono::seconds;
195 using std::map;
196 using std::type_index;
197
198 using Float_seconds = duration<double, seconds::period>;
199
200 // Use a floating point representation for socket lifetime so that it can be printed in floating-point seconds.
201 const Float_seconds lifetime = duration_cast<Float_seconds>(Fine_clock::now() - m_init_time);
202
203 /* Compute the loss rate based based on Dropped vs. Acknowledged bytes. In-flight bytes are
204 * considered as having unknown status for now and are not included. If a Dropped packet is later
205 * acknowledged, it has still already been Dropped, so it will still be only counted as Dropped.
206 * However there is a separate statistic for such late acknowledgments. */
207 double den = double(m_dropped_data_size + m_good_ack_data_size);
208 const double drop_pct
209 = (den == double(0))
210 ? 0
211 : (double(m_dropped_data_size) * double(100) / den);
212 // Compute % of retransmitted vs. all sent DATA packets.
213 den = double(m_sent_data_size);
214 const double rexmit_pct
215 = (den == double(0))
216 ? 0
217 : (double(m_sent_rexmitted_data_size) * double(100) / den);
218
219 *os
220 << "[snd] format of sizes: [bytes|packets]\n"
221
222 "[snd] socket_lifetime [" << lifetime << "]\n"
223
224 "[snd] lifetime_data_sent "
225 "[" << util::to_mbit_per_sec<Float_seconds>(double(m_sent_data_size) / lifetime.count()) << " Mbit/s]\n"
226
227 "[snd] sent_data [" << m_sent_data_size << '|' << m_sent_data_count << "]\n"
228
229 "[snd] acked [" << m_good_ack_data_size << '|' << m_good_ack_count << "] "
230 "dropped [" << m_dropped_data_size << '|' << m_dropped_data_count << "]\n"
231
232 "[snd] observed_net_drop_rate = dropped / (acked + dropped) = [" << drop_pct << "%]\n"
233
234 "[snd] in_flight [" << (m_sent_data_size - m_good_ack_data_size - m_dropped_data_size)
235 << '|' << (m_sent_data_count - m_good_ack_count - m_dropped_data_count) << "]\n"
236
237 "[snd] original [" << (m_sent_data_size - m_sent_rexmitted_data_size)
238 << '|' << (m_sent_data_count - m_sent_rexmitted_data_count) << "] "
239 "rexmitted [" << m_sent_rexmitted_data_size << '|' << m_sent_rexmitted_data_count << "]\n"
240
241 "[snd] rexmit_rate = rexmitted / sent = [" << rexmit_pct << "%]\n"
242
243 "[snd] received_low_lvl_acks [" << m_received_low_lvl_ack_count << " packets]\n"
244
245 "[snd] rcv_wnd_update_only_acks [" << m_received_low_lvl_rcv_wnd_only_ack_count << " packets]\n"
246
247 "[snd] received_acks [" << m_received_ack_count << " individual acknowledgments]\n"
248
249 "[snd] good [" << m_good_ack_count << " individual acknowledgments]\n"
250
251 "[snd] late_or_dupe [" << m_late_or_dupe_ack_count << " individual acknowledgments]\n"
252
253 "[snd] error [" << m_error_acks_data_count << " individual acknowledgments]\n"
254
255 "[snd] loss_events [" << m_loss_events << "]\n"
256
257 "[snd] drop_timeouts [" << m_drop_timeouts << "]\n"
258
259 "[snd] idle_timeouts [" << m_idle_timeouts << "]\n"
260
261 "[snd] remote_rcv_wnd_exhaustion_events [" << m_remote_rcv_wnd_exhaustion_events << "]\n"
262
263 "[snd] recovered [" << m_remote_rcv_wnd_recovery_events << "]\n"
264
265 "[snd] max_buf_data_size [" << m_max_buf_data_size << " bytes]\n"
266 "[snd] low_lvl_packet_send_requested ";
267
268 Peer_socket_info::output_map_of_pkt_counts(os, m_low_lvl_packet_xfer_requested_count_by_type,
269 static_cast<const map<type_index, uint64_t>*>(0));
270
271 *os
272 << "]\n" <<
273 "[snd] send_called ";

325
326 const Fine_duration& srtt = m_snd_smoothed_round_trip_time;

337 "--- Buffers/queues/lists ----\n"

339 " = limit\n"
340 " - Receive buffer size: [" << m_rcv_buf_size << "]\n"
341 " - reassembly queue total data size: [" << m_rcv_reassembly_q_data_size << "] (0 if rexmit off)\n"
342 " " << (m_sock_opts.m_st_rexmit_on ? "Reassembly queue" : "Dupe check list") <<
343 " length: [" << m_rcv_packets_with_gaps << "].\n"

345
346 // This condition is hard to notice otherwise but can explain hard-to-debug refusal to send any data.

355 "DATA packets in SYN_RCVD queue: "

367
368 *os
369 <<
370 "--- Traffic stats (outgoing) ---\n" << m_snd << "\n"
371 "--- Traffic stats (incoming) ---\n" << m_rcv << "\n"

373 "--- Congestion control status ---\n"

406template<typename Key, typename Value>
407void Peer_socket_info::output_map_of_pkt_counts(std::ostream* os,
408 const std::map<Key, Value>& count_by_type,
409 const std::map<Key, Value>* size_by_type) // Static.
410{
411 // Output two maps with the same key set (but 2nd map may not exist, if null).
412 *os << '[';
413
414 size_t idx = 0;
415 for (const auto& type_id_and_count : count_by_type)
416 {
417 // Common key; value from first map.
418 const Key& type_id = type_id_and_count.first;
419 const Value& value = type_id_and_count.second;
420
421 output_pkt_count_key(os, type_id);
422 *os << ":";
423
424 if (size_by_type)
425 {
426 *os << '[';
427
428 // Value from second map. Type can be a simple scalar or something more complex, so use this function.
429 output_pkt_count_value(os, size_by_type->find(type_id)->second);
430 *os << "b|";
431 }
432
433 // Value from first map. Similar to above.
434 output_pkt_count_value(os, value);
435
436 if (size_by_type)
437 {
438 *os << ']';
439 }
440
441 if (idx != count_by_type.size() - 1)
442 {
443 *os << ' ';
444 }
445 ++idx;
446 }
447
448 *os << ']';
449}

456void Peer_socket_info::output_pkt_count_value(std::ostream* os,
457 const boost::array<uint64_t, 2>& value_by_delay_type) // Static.
458{
459 const auto& value_no_delay = value_by_delay_type[0];
460 const auto& value_paced = value_by_delay_type[1];
461 const char* const NO_DELAY = "NO_DELAY";
462 const char* const PACED = "PACED";
463
464 if ((value_no_delay == 0) && (value_paced == 0))
465 {
466 *os << '0';
467 }
468 else if (value_no_delay == 0)
469 {
470 *os << '[' << PACED << ':' << value_paced << ']';
471 }
472 else if (value_paced == 0)
473 {
474 *os << '[' << NO_DELAY << ':' << value_no_delay << ']';
475 }
476 else
477 {
478 *os << '[' << NO_DELAY << ':' << value_no_delay << ' ' << PACED << ':' << value_paced << ']';
479 }
